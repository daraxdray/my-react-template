Javascript

Javascript Structure

All the javascript plugins used by the template are placed at /src/libs/* directory. Every single component JS file are imported in /src/libs/index.js. You can modify this file to add or remove components depends to your needs. Here is the part things you need to know about the /src/libs/index.js structure and functions. 

                         
                            /*
                            |--------------------------------------------------------------------------
                            | Tailwise Built-in Components
                            |--------------------------------------------------------------------------
                            |
                            | Import Tailwise built-in components.
                            |
                            */
                            import "@left4code/tw-starter/dist/js/svg-loader";
                            import "@left4code/tw-starter/dist/js/accordion";
                            import "@left4code/tw-starter/dist/js/alert";
                            import "@left4code/tw-starter/dist/js/dropdown";
                            import "@left4code/tw-starter/dist/js/modal";
                            import "@left4code/tw-starter/dist/js/tab";

                            /*
                            |--------------------------------------------------------------------------
                            | Custom Components
                            |--------------------------------------------------------------------------
                            |
                            | Import JS custom components.
                            |
                            */
                            import "./show-code";
                            import "./copy-code";
                            import "./zoom";
